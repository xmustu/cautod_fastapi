```python
import cadquery as cq

def create_gear(module, teeth):
    """
    使用 CadQuery 创建一个简单的齿轮。
    """
    # 参数
    pressure_angle = 20.0
    
    # 创建齿轮
    gear = cq.Workplane("XY").gear(
        module=module, 
        teeth=teeth, 
        pressure_angle=pressure_angle
    )
    
    return gear

# 创建一个齿轮实例
result = create_gear(module=1.0, teeth=30)

# 显示或保存模型
# show_object(result)
cq.exporters.export(result, 'gear.step')

print("齿轮模型已生成并保存为 gear.step")
```
